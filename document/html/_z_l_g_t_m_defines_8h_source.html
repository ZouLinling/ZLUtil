111 // _GTMDevLog & _GTMDevAssert
112 //
113 // _GTMDevLog & _GTMDevAssert are meant to be a very lightweight shell for
114 // developer level errors. This implementation simply macros to NSLog/NSAssert.
115 // It is not intended to be a general logging/reporting system.
116 //
117 // Please see http://code.google.com/p/google-toolbox-for-mac/wiki/DevLogNAssert
118 // for a little more background on the usage of these macros.
119 //
120 // _GTMDevLog log some error/problem in debug builds
121 // _GTMDevAssert assert if conditon isn't met w/in a method/function
122 // in all builds.
123 //
124 // To replace this system, just provide different macro definitions in your
125 // prefix header. Remember, any implementation you provide *must* be thread
126 // safe since this could be called by anything in what ever situtation it has
127 // been placed in.
128 //
129 
130 // We only define the simple macros if nothing else has defined this.
131 #ifndef _GTMDevLog
132 
133 #ifdef DEBUG
134 #define _GTMDevLog(...) NSLog(__VA_ARGS__)
135 #else
136 #define _GTMDevLog(...) do { } while (0)
137 #endif
138 
139 #endif // _GTMDevLog

141 #ifndef _GTMDevAssert
142 // we directly invoke the NSAssert handler so we can pass on the varargs
143 // (NSAssert doesn't have a macro we can use that takes varargs)
144 #if !defined(NS_BLOCK_ASSERTIONS)
145 #define _GTMDevAssert(condition, ...) \
146 do { \
147 if (!(condition)) { \
148 [[NSAssertionHandler currentHandler] \
149 handleFailureInFunction:[NSString stringWithUTF8String:__PRETTY_FUNCTION__] \
150 file:[NSString stringWithUTF8String:__FILE__] \
151 lineNumber:__LINE__ \
152 description:__VA_ARGS__]; \
153 } \
154 } while(0)
155 #else // !defined(NS_BLOCK_ASSERTIONS)
156 #define _GTMDevAssert(condition, ...) do { } while (0)
157 #endif // !defined(NS_BLOCK_ASSERTIONS)
158 
159 #endif // _GTMDevAssert

161 // _GTMCompileAssert
162 // _GTMCompileAssert is an assert that is meant to fire at compile time if you
163 // want to check things at compile instead of runtime. For example if you
164 // want to check that a wchar is 4 bytes instead of 2 you would use
165 // _GTMCompileAssert(sizeof(wchar_t) == 4, wchar_t_is_4_bytes_on_OS_X)
166 // Note that the second "arg" is not in quotes, and must be a valid processor
167 // symbol in it's own right (no spaces, punctuation etc).
168 
169 // Wrapping this in an #ifndef allows external groups to define their own
170 // compile time assert scheme.
171 #ifndef _GTMCompileAssert
172 // We got this technique from here:
173 // http://unixjunkie.blogspot.com/2007/10/better-compile-time-asserts_29.html
174 
175 #define _GTMCompileAssertSymbolInner(line, msg) _GTMCOMPILEASSERT ## line ## __ ## msg
176 #define _GTMCompileAssertSymbol(line, msg) _GTMCompileAssertSymbolInner(line, msg)
177 #define _GTMCompileAssert(test, msg) \
178 typedef char _GTMCompileAssertSymbol(__LINE__, msg) [ ((test) ? 1 : -1) ]
179 #endif // _GTMCompileAssert

370 // Macro to allow fast enumeration when building for 10.5 or later, and
371 // reliance on NSEnumerator for 10.4. Remember, NSDictionary w/ FastEnumeration
372 // does keys, so pick the right thing, nothing is done on the FastEnumeration
373 // side to be sure you're getting what you wanted.
374 #ifndef GTM_FOREACH_OBJECT
375 #if TARGET_OS_IPHONE || !(MAC_OS_X_VERSION_MIN_REQUIRED < MAC_OS_X_VERSION_10_5)
376 #define GTM_FOREACH_ENUMEREE(element, enumeration) \
377 for (element in enumeration)
378 #define GTM_FOREACH_OBJECT(element, collection) \
379 for (element in collection)
380 #define GTM_FOREACH_KEY(element, collection) \
381 for (element in collection)
382 #else
383 #define GTM_FOREACH_ENUMEREE(element, enumeration) \
384 for (NSEnumerator *_ ## element ## _enum = enumeration; \
385 (element = [_ ## element ## _enum nextObject]) != nil; )
386 #define GTM_FOREACH_OBJECT(element, collection) \
387 GTM_FOREACH_ENUMEREE(element, [collection objectEnumerator])

389 GTM_FOREACH_ENUMEREE(element, [collection keyEnumerator])
